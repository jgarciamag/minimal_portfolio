def read_excel_sheet(file_path, sheet_name):
    df = pd.read_excel(file_path, sheet_name=sheet_name)
    df.set_index('Date', inplace=True) 
    df.rename(columns={'Price': sheet_name}, inplace=True)
    
    # Filter data from the last value of 2023 until the current date
    last_date_2023 = df[df.index.year == 2023].index.max()
    df = df[(df.index >= last_date_2023)]
    
    df['Percent Change'] = df[sheet_name].pct_change() * 100
    
    # Cumulative return
    df['Cumulative Return'] = (1 + df['Percent Change'] / 100).cumprod() * 100
    df['Cumulative Return'].iloc[0] = 100  # Set the first value to 100

    # Calculate Drawdown
    df['Drawdown'] = (df['Cumulative Return'] - df['Cumulative Return'].expanding().max()) / df['Cumulative Return'].expanding().max()
    df['Drawdown'] = df['Drawdown'].clip(upper=0)  # Clip values above 0 to 0
    
    return df

def plot_drawdown(df, df_name, benchmark=None, benchmark_name=None):
    plt.figure(figsize=(18, 5))
    
    #Plot Drawdown
    plt.fill_between(df.index, df['Drawdown'], color='lightblue', alpha=0.7, label=f'{df_name} Drawdown')
    
    #Plot Benchmark 
    if benchmark is not None:
        plt.fill_between(benchmark.index, benchmark['Drawdown'], color='green', alpha=0.7, label=f'{benchmark_name} Drawdown')
    
    # Set title and labels
    title = f'{df_name} Drawdown Analysis'
    plt.title(title)
    plt.xlabel('Date')
    plt.ylabel('Drawdown (%)')
    plt.legend()
    plt.grid(True)
    plt.show()

from dateutil.relativedelta import relativedelta

def yearfrac(start_date, end_date):
    # Function to calculate the fraction of a year between two dates
    diff = relativedelta(end_date, start_date)
    year_fraction = diff.years + (diff.months / 12) + (diff.days / 365.25)
    return year_fraction

def cgar(first_value, last_value, fracc):
    # Function to calculate the compound annual growth rate (CAGR)
    return (last_value / first_value) ** (1 / fracc) - 1

def sharpe_ratio(file_path, sheet_name):
    df = pd.read_excel(file_path, sheet_name=sheet_name)
    df.set_index('Date', inplace=True)
    
    df.index = pd.to_datetime(df.index)
    
    # Filter the data to show from 2017 onwards
    #df = df[df.index.year >= 2017]

    df = df.tail(252)
    
    first_index = df.index[0]
    last_index = df.index[-1]
    fracc = yearfrac(first_index, last_index)
    
    first_value = df.iloc[0, 0]
    last_value = df.iloc[-1, 0]
    
    cagr = cgar(first_value, last_value, fracc)
    
    std = df.iloc[:, 0].std(ddof=1)
    
    risk_free_rate = 5.33 / 100
    
    sharpe_ratio = (cagr - risk_free_rate) / std
    
    df.rename(columns={'Price': sheet_name}, inplace=True)
    
    return df, fracc, cagr, std, sharpe_ratio

from dateutil.relativedelta import relativedelta

def yearfrac(start_date, end_date):
    # Function to calculate the fraction of a year between two dates
    diff = relativedelta(end_date, start_date)
    year_fraction = diff.years + (diff.months / 12) + (diff.days / 365.25)
    return year_fraction

def cgar(first_value, last_value, fracc):
    # Function to calculate the compound annual growth rate (CAGR)
    return (last_value / first_value) ** (1 / fracc) - 1

def sharpe_ratio_and_beta(file_path, sheet_name1, sheet_name2):
    df1 = pd.read_excel(file_path, sheet_name=sheet_name1, index_col='Date')
    df2 = pd.read_excel(file_path, sheet_name=sheet_name2, index_col='Date')
    
    # Filter the data to the last 252 trading days
    df1 = df1.tail(252)
    df2 = df2.tail(252)
    
    first_index1 = df1.index[0]
    last_index1 = df1.index[-1]
    fracc1 = yearfrac(first_index1, last_index1)
    
    first_index2 = df2.index[0]
    last_index2 = df2.index[-1]
    fracc2 = yearfrac(first_index2, last_index2)
    
    first_value1 = df1.iloc[0, 0]
    last_value1 = df1.iloc[-1, 0]
    
    first_value2 = df2.iloc[0, 0]
    last_value2 = df2.iloc[-1, 0]
    
    cagr1 = cgar(first_value1, last_value1, fracc1)
    cagr2 = cgar(first_value2, last_value2, fracc2)
    
    std1 = df1.iloc[:, 0].std(ddof=1)
    std2 = df2.iloc[:, 0].std(ddof=1)
    
    var1 = df1.iloc[:, 0].var()
    var2 = df2.iloc[:, 0].var()
    
    cov = df1.iloc[:, 0].cov(df2.iloc[:, 0])
    
    beta = cov / var2
    
    # Define the risk-free rate
    risk_free_rate = 5.33 / 100
    
    jensen_alpha = cagr1 - (risk_free_rate + (cagr2 - risk_free_rate) * beta)
    
    tracking_error = ((var1 - (beta ** 2 * var2)) ** 0.5)
    
    information_ratio = jensen_alpha / tracking_error
    
    sharpe_ratio1 = (cagr1 - risk_free_rate) / std1
    sharpe_ratio2 = (cagr2 - risk_free_rate) / std2
    
    df1.rename(columns={'Price': sheet_name1}, inplace=True)
    df2.rename(columns={'Price': sheet_name2}, inplace=True)
    
    return df1, fracc1, cagr1, std1, sharpe_ratio1, df2, fracc2, cagr2, std2, sharpe_ratio2, beta, jensen_alpha, tracking_error, information_ratio
